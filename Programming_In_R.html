htru2_sum <- function(htru2_df, grouped=TRUE, which_vars, which_fun=mean) {
  if(grouped) {
    htru2_df %>% 
      group_by(Class) %>% 
      summarise_at(which_vars,which_fun)
  }
  else {
    htru2_df %>% 
      summarise_at(which_vars,which_fun)
  } 
}
